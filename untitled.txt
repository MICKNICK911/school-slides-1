    <script>
        let currentPosition = 0;
        let isUppercase = true;
        const slider = document.getElementById('slider');
        const slides = document.querySelectorAll('.screen');
        const letterElements = document.querySelectorAll('.letter');
        const totalSlides = slides.length - 1;

        function initSlider() {
            slider.style.height = `${slides.length * 100}vh`;
        }
        initSlider();

        function toggleTheme() {
            document.body.classList.toggle('dark-mode');
            localStorage.setItem('theme', 
                document.body.classList.contains('dark-mode') ? 'dark' : 'light');
        }

        function toggleCase() {
            isUppercase = !isUppercase;
            localStorage.setItem('letterCase', isUppercase ? 'upper' : 'lower');
            updateLetterCase();
        }

        function updateLetterCase() {
            letterElements.forEach(el => {
                el.textContent = isUppercase ? el.dataset.upper : el.dataset.lower;
            });
        }

        if (localStorage.getItem('theme') === 'dark') {
            document.body.classList.add('dark-mode');
        }
        if (localStorage.getItem('letterCase') === 'lower') {
            isUppercase = false;
            updateLetterCase();
        }

        function navigate(position) {
            currentPosition = Math.max(0, Math.min(position, totalSlides));
            slider.style.transform = `translateY(-${currentPosition * 100}vh)`;
        }

        function nextSlide() {
            navigate(currentPosition + 1);
        }

        function prevSlide() {
            navigate(currentPosition - 1);
        }

        function instantRestart() {
            slider.style.transition = 'none';
            currentPosition = 0;
            slider.style.transform = 'translateY(0)';
            void slider.offsetHeight;
            slider.style.transition = '';
        }

        let startY = 0;
        const swipeThreshold = 50;

        document.addEventListener('touchstart', e => {
            startY = e.touches[0].clientY;
        });

        document.addEventListener('touchend', handleSwipe);
        document.addEventListener('mousedown', e => {
            startY = e.clientY;
        });
        document.addEventListener('mouseup', handleSwipe);

        function handleSwipe(e) {
            const endY = e.changedTouches?.[0].clientY || e.clientY;
            const deltaY = endY - startY;

            if (Math.abs(deltaY) > swipeThreshold) {
                deltaY > 0 ? prevSlide() : nextSlide();
            }
        }

        window.addEventListener('load', () => {
            setTimeout(() => window.scrollTo(0, 1), 0);
            updateLetterCase();
        });
        window.addEventListener('resize', () => setTimeout(() => window.scrollTo(0, 1), 0));
        
        
        document.querySelectorAll('.letter').forEach(letter => {
            letter.addEventListener('click', function() {
                this.classList.toggle('clicked');
            });
        });
    </script>
